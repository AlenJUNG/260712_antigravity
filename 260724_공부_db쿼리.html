DAY 1 데이터 조회의 첫걸음

SELECT · WHERE · ORDER BY — 조회의 뼈대는 SELECT 컬럼 FROM 테이블 WHERE 조건 ORDER BY 정렬

초급SELECT와 WHERE 기본

전체 / 특정 컬럼 조회

SELECT * FROM customers;              -- 모든 컬럼
SELECT name, city FROM customers;   -- 원하는 컬럼만 (실무 권장)

WHERE로 조건 걸기

SELECT name, city FROM customers WHERE city = '서울';   -- 문자열은 작은따옴표
SELECT order_id, amount FROM orders WHERE amount >= 50000; -- 숫자는 따옴표 X

비교 연산자: =, <>(같지 않다), >, <, >=, <=

연습 · '전자' 카테고리 상품의 이름과 가격을 조회하라 
SELECT product_name, price FROM products WHERE category = '전자';
중급여러 조건 조합하기

AND / OR / NOT

SELECT * FROM orders WHERE status = 'PAID' AND amount >= 50000;
SELECT * FROM customers WHERE city = '서울' OR city = '부산';
⚠️ 우선순위 — AND가 OR보다 먼저 계산됨. 헷갈리면 괄호로 명시: (a OR b) AND c

BETWEEN / IN / LIKE

SELECT * FROM orders WHERE amount BETWEEN 30000 AND 100000;  -- 양끝 포함
SELECT * FROM customers WHERE city IN ('서울','부산');       -- OR을 깔끔하게
SELECT * FROM products WHERE product_name LIKE '%키보드%'; -- 포함
SELECT * FROM customers WHERE name LIKE '김%';           -- 시작

% = 아무 문자 0개 이상, _ = 아무 문자 딱 1개

연습 · ①20,000~90,000원 주문 ②'박'으로 시작하는 고객 
SELECT * FROM orders WHERE amount BETWEEN 20000 AND 90000;
SELECT * FROM customers WHERE name LIKE '박%';
고급정렬 · 개수 제한 · 중복 제거 · NULL
-- 다중 정렬
SELECT name, city, signup_date FROM customers ORDER BY city ASC, signup_date DESC;
-- 상위 N개 / 페이징
SELECT order_id, amount FROM orders ORDER BY amount DESC LIMIT 3;
SELECT * FROM orders ORDER BY order_id LIMIT 3 OFFSET 3;   -- 4~6번째
-- 중복 제거 / NULL
SELECT DISTINCT city FROM customers;
SELECT * FROM orders WHERE status IS NULL;   -- = NULL 아님!
⚠️ NULL 함정 — = NULL은 항상 거짓. 반드시 IS NULL / IS NOT NULL 사용.
📌 DBMS 차이 — 상위 N개: SQL Server TOP 3, Oracle FETCH FIRST 3 ROWS ONLY
연습 · ①금액 상위 2건 ②고객 도시 종류(중복 없이) 
SELECT * FROM orders ORDER BY amount DESC LIMIT 2;
SELECT DISTINCT city FROM customers;

DAY 2 함수와 데이터 가공

저장된 값을 그대로 보는 데서 나아가, 함수로 가공·계산해 원하는 형태로 뽑는다.

초급산술 연산과 별칭(AS)
SELECT product_name, price, price * 1.1 AS price_with_vat FROM products;  -- 부가세 10%
SELECT CONCAT(name, '(', city, ')') AS label FROM customers;         -- 김서준(서울)
📌 DBMS 차이 — Oracle·PostgreSQL은 name || '(' || city || ')' (|| 연결)도 가능
연습 · 배송비 3,000원 뺀 값을 net_amount로 
SELECT order_id, amount, amount - 3000 AS net_amount FROM orders;
중급문자열 · 숫자 함수
분류함수결과 예시
문자열UPPER / LOWER'sql' → 'SQL' / 'SQL' → 'sql'
LENGTH(x)글자 수
SUBSTRING(x,1,1)첫 글자(성)
REPLACE / TRIM치환 / 앞뒤 공백 제거
숫자ROUND(12345.6,0)12346
CEIL(12.1)13
FLOOR(12.9)12

연습 · 이름 첫 글자(성)와 글자 수를 함께 
SELECT name, SUBSTRING(name,1,1) AS surname, LENGTH(name) AS len FROM customers;
고급날짜 함수 · CASE · NULL 함수

날짜 다루기 & CASE WHEN

SELECT order_id, amount,
    CASE
        WHEN amount >= 100000 THEN 'VIP'
        WHEN amount >= 50000  THEN '일반'
        ELSE '소액'
    END AS grade,
    YEAR(order_date) AS y, MONTH(order_date) AS m
FROM orders;

CASE는 if-else처럼 위에서부터 처음 맞는 조건을 채택합니다.

SELECT order_id, COALESCE(status, '미정') AS status FROM orders;  -- NULL이면 '미정'
📌 DBMS 차이 — 날짜 추출: PostgreSQL EXTRACT(YEAR FROM order_date) · 날짜 차이: MySQL DATEDIFF(a,b), PostgreSQL a - b
연습 · 금액별 VIP(10만↑)/일반/소액(5만↓) 분류 
SELECT order_id, amount,
    CASE WHEN amount>=100000 THEN 'VIP'
         WHEN amount>=50000 THEN '일반' ELSE '소액' END AS grade
FROM orders;

DAY 3 집계와 그룹화

여러 행을 하나로 요약하는 것이 집계. "무엇을 기준으로 묶을지"가 GROUP BY.

초급집계 함수 5종
SELECT COUNT(*) AS 주문건수, SUM(amount) AS 총매출,
       AVG(amount) AS 평균금액, MAX(amount) AS 최고, MIN(amount) AS 최저
FROM orders;

COUNT(*)는 행 수, COUNT(컬럼)은 NULL이 아닌 값의 수.

연습 · PAID 주문의 건수와 총매출 
SELECT COUNT(*) AS cnt, SUM(amount) AS total FROM orders WHERE status='PAID';
중급GROUP BY로 묶어서 집계
SELECT customer_id, COUNT(*) AS 주문수, SUM(amount) AS 총결제액
FROM orders GROUP BY customer_id;
⚠️ 규칙 — SELECT에 오는 컬럼은 집계 함수이거나 GROUP BY에 있어야 합니다.
연습 · 주문 상태(status)별 건수 
SELECT status, COUNT(*) AS cnt FROM orders GROUP BY status;
고급HAVING · WHERE vs HAVING
SELECT customer_id, SUM(amount) AS 총결제액
FROM orders
WHERE status = 'PAID'            -- ① 개별 행 필터 (집계 전)
GROUP BY customer_id
HAVING SUM(amount) >= 100000;   -- ② 집계 결과 필터 (집계 후)
구분시점조건 대상집계 함수
WHERE그룹 묶기 전 (②)개별 행사용 불가
HAVING그룹 묶은 후 (④)집계 결과SUM,COUNT 가능
연습 · 주문 2건 이상인 고객의 id와 주문 수 
SELECT customer_id, COUNT(*) AS cnt
FROM orders GROUP BY customer_id HAVING COUNT(*) >= 2;

DAY 4 여러 테이블 다루기 (JOIN)

흩어진 테이블을 공통 키로 연결해 하나의 결과로 합치는 것이 JOIN.

활용 — customers LEFT JOIN orders: 주문 없는 정유나도 나오고 주문 열은 NULL. 그래서 "주문 안 한 고객 찾기"는 LEFT JOIN … WHERE order_id IS NULL 패턴.
초급INNER JOIN
-- 주문 정보에 고객 이름 붙이기
SELECT o.order_id, c.name, o.amount
FROM orders o
INNER JOIN customers c ON o.customer_id = c.customer_id;

o, c는 테이블 별칭. ON에 연결 기준. INNER JOIN은 양쪽에 다 있는 행만(주문 없는 정유나 제외).

연습 · order_items에 상품명 붙이기 
SELECT oi.order_id, p.product_name, oi.quantity
FROM order_items oi JOIN products p ON oi.product_id = p.product_id;
중급LEFT JOIN과 3중 조인
-- 모든 고객 나열 (주문 없어도 고객은 나옴)
SELECT c.name, o.order_id, o.amount
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id;

-- 세 테이블 연결: 주문 → 주문상세 → 상품
SELECT c.name, o.order_id, p.product_name, oi.quantity
FROM orders o
JOIN customers   c  ON o.customer_id = c.customer_id
JOIN order_items oi ON o.order_id    = oi.order_id
JOIN products    p  ON oi.product_id = p.product_id;
연습 · 주문이 없는 고객의 이름 (LEFT JOIN + IS NULL) 
SELECT c.name FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;
고급SELF JOIN · UNION · JOIN + 집계

JOIN + 집계 (가장 실무적)

SELECT c.city, SUM(o.amount) AS 도시매출
FROM customers c JOIN orders o ON c.customer_id = o.customer_id
WHERE o.status = 'PAID'
GROUP BY c.city ORDER BY 도시매출 DESC;

UNION · SELF JOIN

SELECT name AS 이름, '고객' AS 구분 FROM customers
UNION ALL
SELECT product_name, '상품' FROM products;   -- 위아래 컬럼 수·타입 같아야 함

-- 같은 도시에 사는 고객 짝 찾기 (자기 자신과 조인)
SELECT a.name AS 고객A, b.name AS 고객B, a.city
FROM customers a JOIN customers b
  ON a.city = b.city AND a.customer_id < b.customer_id;
UNION vs UNION ALL — UNION은 중복 제거(느림), UNION ALL은 중복 포함(빠름).
연습 · 고객별 총결제액(PAID), 주문 없으면 0으로 
SELECT c.name, COALESCE(SUM(o.amount), 0) AS total
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id AND o.status='PAID'
GROUP BY c.name;
-- 힌트: status를 ON에 두어야 LEFT JOIN 유지 (WHERE에 두면 0원 고객이 사라짐)

DAY 5 서브쿼리에서 윈도우 함수까지

"쿼리 안에 쿼리를 넣는 법(서브쿼리)"과 "행을 유지한 채 순위·누적·비교를 계산하는 법(윈도우 함수)".

초급서브쿼리의 기본

서브쿼리 = "먼저 계산해야 할 값"을 괄호 () 안에 넣어 미리 구하는 것.

-- 전체 평균보다 큰 주문
SELECT order_id, amount FROM orders WHERE amount > (SELECT AVG(amount) FROM orders);
-- 서울 고객의 주문 (다중행 → IN)
SELECT order_id, amount FROM orders
WHERE customer_id IN (SELECT customer_id FROM customers WHERE city='서울');
연습 · ①평균보다 싼 상품 ②부산 고객의 주문 수 
SELECT product_name, price FROM products WHERE price < (SELECT AVG(price) FROM products);
SELECT COUNT(*) FROM orders WHERE customer_id IN (SELECT customer_id FROM customers WHERE city='부산');
중급상관 서브쿼리 · 인라인 뷰 · EXISTS · CTE
기법위치특징
상관 서브쿼리WHERE/SELECT바깥 행 값을 참조, 행마다 재실행
인라인 뷰FROM서브쿼리 결과를 임시 테이블처럼
EXISTSWHERE값이 아니라 "존재 여부"만 → 빠름
CTE (WITH)맨 앞이름 붙여 단계별로 쪼갬
 
-- 상관 서브쿼리: 그 고객 '자신의' 평균보다 큰 주문 (고객마다 기준이 다름)
SELECT o.order_id, o.customer_id, o.amount FROM orders o
WHERE o.amount > (SELECT AVG(o2.amount) FROM orders o2 WHERE o2.customer_id = o.customer_id);

-- EXISTS: 주문을 한 번이라도 한 고객
SELECT c.customer_id, c.name FROM customers c
WHERE EXISTS (SELECT 1 FROM orders o WHERE o.customer_id = c.customer_id);

-- CTE: 서브쿼리에 이름을 붙여 가독성 UP
WITH customer_totals AS (
    SELECT customer_id, SUM(amount) AS total FROM orders WHERE status='PAID' GROUP BY customer_id
)
SELECT * FROM customer_totals WHERE total >= 100000 ORDER BY total DESC;
SELECT 1인 이유: EXISTS는 존재 여부만 보므로 무엇을 SELECT하든 무관. NOT EXISTS → "주문한 적 없는 고객".
연습 · ①PAID 주문 있는 고객(EXISTS) ②고객 2명↑ 도시(인라인 뷰) 
SELECT c.customer_id, c.name FROM customers c
WHERE EXISTS (SELECT 1 FROM orders o WHERE o.customer_id=c.customer_id AND o.status='PAID');

SELECT city, cnt FROM (
    SELECT city, COUNT(*) AS cnt FROM customers GROUP BY city
) t WHERE cnt >= 2;
고급윈도우 함수 · 재귀 CTE

GROUP BY는 행을 합쳐 없애지만, 윈도우 함수는 행을 그대로 두고 옆에 계산 결과를 붙입니다.

순위 함수 3형제 (동점 처리 차이)

SELECT order_id, amount,
    ROW_NUMBER() OVER (ORDER BY amount DESC) AS row_num,
    RANK()       OVER (ORDER BY amount DESC) AS rnk,
    DENSE_RANK() OVER (ORDER BY amount DESC) AS dense
FROM orders;
함수동점일 때 (예: 100,90,90,80)
ROW_NUMBER1, 2, 3, 4  (무조건 순번)
RANK1, 2, 2, 4  (동점 후 건너뜀)
DENSE_RANK1, 2, 2, 3  (동점 후 안 건너뜀)

그룹별 순위 · 이동 · 누적

-- 고객별 최고 금액 주문 (그룹별 TOP N 정석 패턴)
WITH ranked AS (
  SELECT customer_id, order_id, amount,
    ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY amount DESC) AS rn
  FROM orders )
SELECT * FROM ranked WHERE rn = 1;

-- LAG: 직전 주문 대비 증감 / SUM OVER: 누적합
SELECT order_date, amount,
    LAG(amount) OVER (ORDER BY order_date) AS 직전,
    SUM(amount) OVER (ORDER BY order_date) AS 누적합
FROM orders WHERE status='PAID';

재귀 CTE (계층·연속 데이터 생성)

WITH RECURSIVE numbers AS (
    SELECT 1 AS n                 -- ① 시작(앵커)
    UNION ALL
    SELECT n + 1 FROM numbers      -- ② 자기 참조로 반복
    WHERE n < 5                    -- ③ 종료 조건 (필수!)
)
SELECT n FROM numbers;                 -- 1,2,3,4,5
📌 DBMS 차이 — SQL Server는 WITH(RECURSIVE 없음), Oracle은 CONNECT BY도 지원. 실무: 조직도·카테고리 트리·빈 날짜 채우기.
연습 · ①category별 가격 DENSE_RANK ②고객별 누적 결제금액 
SELECT category, product_name, price,
    DENSE_RANK() OVER (PARTITION BY category ORDER BY price DESC) AS price_rank
FROM products;

SELECT customer_id, order_date, amount,
    SUM(amount) OVER (PARTITION BY customer_id ORDER BY order_date) AS cumulative
FROM orders WHERE status='PAID';
